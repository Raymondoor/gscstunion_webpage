CREATE TABLE user (
    id       INTEGER NOT NULL
                     PRIMARY KEY,
    username TEXT    NOT NULL,
    password TEXT    NOT NULL
);
CREATE TABLE article (
    id      INTEGER PRIMARY KEY AUTOINCREMENT
                    UNIQUE
                    NOT NULL,
    title   TEXT    NOT NULL,
    user_id INTEGER NOT NULL
                    REFERENCES user (id) ON DELETE NO ACTION
                                         ON UPDATE NO ACTION
                                         MATCH SIMPLE,
    date    TEXT    NOT NULL,
    main    TEXT    NOT NULL,
    url     TEXT,
    media   BLOB
);
